  • ¶

    Function to determine whether a user is loggedIn. If the user is logged in, the userâ€™s information will be stored in the global angular $rootScope to personalize each view.

        checkLoggedIn   : function () {
      return $http({
        method: 'GET',
        url: '/api/loggedin'
      })
      .then(function (user) {
        if (user.data[0].id) {
          $rootScope.loggedinUserInfo = user.data[0];
          return true;
        } else {
          console.log("error authenticating user");
          return false;
        }
      });

  • ¶

    On

  • ¶

    Socket listener

        on: function(eventName, callback) {
      window.socket.on(eventName, function() {

        var args = arguments;
        $rootScope.$apply(function() {
          callback.apply(window.socket, args);
        })
      });
    },
  • ¶

    Emit

  • ¶

    Socket emitter

        emit: function(eventName, data, callback) {
      window.socket.emit(eventName, data, function() {
        var args = arguments;
        $rootScope.$apply(function() {
          if (callback) {
            callback.apply(window.socket, args);
          }
        })
      });
    }
  }

});
